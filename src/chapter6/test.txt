int getlines(char s[], int lim){
    int c, i;

 

    for (i=0; (c=getchar()) !=EOF && c!='q' && i < lim; ++i)    
        s[i] = c;
    
    if (c == '\n') 
    {
        s[i] = c;
        ++i;
    }

    s[i] = '\0';

    if ( i >= lim)
        i = -1;

    return i;
}
